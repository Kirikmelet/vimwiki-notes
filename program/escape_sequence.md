= Escape Sequences =
== 2020-02-24 ==
=== [[Programming_Notes|Index]] ===

"b" = Backspace

"f" = Form Feed

"n" = Newline

"r" = Return

"t" = Horizontal tab

"v" = Vertical Tab

"" = Backslash

"SINGLE QOUTE HERE" = Single Quote

"DOUBLE QOUTES HERE" = Double Quote

"?" = Question Mark

"0" = NULL
